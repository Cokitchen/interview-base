import { Multiselect } from './multiselect';

A component that does something special and renders text in a div.

### Component usage

```js
const [values, setValues] = useState(['potato'])

<Multiselect
  dropdownPlaceholder="Select all or multiple brands"
  placeholder="Select brands"
  values={values}
  onChange={values => setValues(values)}
  options={[
    {
      text: 'Potato',
      value: 'potato',
    },
    {
      text: 'Salad Cream',
      value: 'salad',
    },
    {
      text: 'Custard Pie',
      value: 'custard',
    },
    {
      text: 'Premium cake',
      value: 'premium_cake',
    },
  ]}
/>
```
