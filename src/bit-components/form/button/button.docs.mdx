import { Button } from './button';

### Component usage

```js
  <Button color="primary"
    variant="block"
    // any other html input prop is allowed
  >
    Click me
  </Button>
/>
```

### Render hello world!

```js live
<Button
  color="primary"
  variant="block"
  onClick={() => alert('You clicked me!')}
>
  Click me
</Button>
```
